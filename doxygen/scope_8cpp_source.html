277 // Scales data based upon desired settings for the channel
278 void
279 Scope::setData(double data[], size_t size)
280 {
281  if (isPaused)
282  return;
283 
284  if (size < getChannelCount()) {
285  ERROR_MSG("Scope::setData() : data size mismatch detected\n");
286  return;
287  }
288 
289  size_t index = 0;
290  for (std::list<Channel>::iterator i = channels.begin(), end = channels.end();
291  i != end;
292  ++i) {
293  i->data[data_idx] = data[index++];
294 
295  if (triggering && i == triggerChannel &&
296  ((triggerDirection == POS && i->data[data_idx - 1] < triggerThreshold &&
297  i->data[data_idx] > triggerThreshold) ||
298  (triggerDirection == NEG && i->data[data_idx - 1] > triggerThreshold &&
299  i->data[data_idx] < triggerThreshold))) {
300  if (data_idx > triggerWindow * fs)
301  triggerQueue.push_back(data_idx - (triggerWindow * fs));
302  }
303  }
304 
305  ++data_idx %= data_size;
306 
307  if (triggering && !triggerQueue.empty() &&
308  (data_idx + 2) % data_size == triggerQueue.front()) {
309  triggerQueue.pop_front();
310  drawCurves();
311  }
312 }

357 void
358 Scope::setTrigger(trig_t direction,
359  double threshold,
360  std::list<Channel>::iterator channel,
361  double window)
362 {
363  if (triggerChannel != channel || triggerThreshold != threshold) {
364  triggerChannel = channel;
365  triggerThreshold = threshold;
366  }
367 
368  // Update if direction has changed
369  if (triggerDirection != direction) {
370  if (direction == Scope::NONE) {
371  triggering = false;
372  timer->start(refresh);
373  triggerQueue.clear();
374  } else {
375  triggering = true;
376  timer->stop();
377  }
378  triggerDirection = direction;
379  }
380  triggerWindow = window;
381 }

469 // Draw data on the scope
470 void
471 Scope::drawCurves(void)
472 {
473  if (isPaused)
474  return;
475 
476  // Set X scale map is same for all channels
477  scaleMapX->setScaleInterval(0, hScl * divX);
478  for (std::list<Channel>::iterator i = channels.begin(), iend = channels.end();
479  i != iend;
480  ++i) {
481  // Set data for channel
482  std::vector<double> x(i->data.size());
483  std::vector<double> y(i->data.size());
484  double* x_loc = x.data();
485  double* y_loc = y.data();
486 
487  // Set Y scale map for channel
488  // TODO this should not happen each iteration, instead build into channel
489  // struct
490  scaleMapY->setScaleInterval(-i->scale * divY / 2, i->scale * divY / 2);
491 
492  // Scale data to pixel coordinates
493  for (size_t j = 0; j < i->data.size(); ++j) {
494  *x_loc = scaleMapX->transform(j * period);
495  *y_loc = scaleMapY->transform(i->data[(data_idx + j) % i->data.size()] +
496  i->offset);
497  ++x_loc;
498  ++y_loc;
499  }
500 
501  // Append data to curve
502  // Makes deep copy - which is not optimal
503  // TODO: change to pointer based method
504  i->curve->setSamples(x.data(), y.data(), i->data.size());
505  }
506 
507  // Update plot
508  replot();
509 }
