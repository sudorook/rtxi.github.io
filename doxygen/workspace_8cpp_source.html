73 Workspace::Instance::Instance(std::string name,
74  Workspace::variable_t* d,
75  size_t n)
76  : IO::Block(name, d, n)
77 {
78  size_t count[] = { 0, 0, 0, 0 };
79  for (size_t i = 0; i < n; ++i)
80  switch (d[i].flags & (PARAMETER | STATE | EVENT | COMMENT)) {
81  case PARAMETER:
82  count[0]++;
83  break;
84  case STATE:
85  count[1]++;
86  break;
87  case EVENT:
88  count[2]++;
89  break;
90  case COMMENT:
91  count[3]++;
92  break;
93  }
94 
95  parameter = std::vector<var_t>(count[0]);
96  state = std::vector<var_t>(count[1]);
97  event = std::vector<var_t>(count[2]);
98  comment = std::vector<comment_t>(count[3]);
99 
100  size_t i[] = { 0, 0, 0, 0 };
101  for (size_t j = 0; j < n; ++j) {
102  switch (d[j].flags & (PARAMETER | STATE | EVENT | COMMENT)) {
103  case PARAMETER:
104  parameter[i[0]].name = d[j].name;
105  parameter[i[0]].description = d[j].description;
106  parameter[i[0]].data = new double;
107  i[0]++;
108  break;
109  case STATE:
110  state[i[1]].name = d[j].name;
111  state[i[1]].description = d[j].description;
112  state[i[1]].data = 0;
113  i[1]++;
114  break;
115  case EVENT:
116  event[i[2]].name = d[j].name;
117  event[i[2]].description = d[j].description;
118  event[i[2]].data = 0;
119  i[2]++;
120  break;
121  case COMMENT:
122  comment[i[3]].name = d[j].name;
123  comment[i[3]].description = d[j].description;
124  comment[i[3]].comment = "";
125  i[3]++;
126  break;
127  }
128  }
129 
130  Workspace::Manager::getInstance()->insertWorkspace(this);
131 }

193 double
194 Workspace::Instance::getValue(IO::flags_t type, size_t n) const
195 {
196  if (type & INPUT)
197  return input(n);
198  if (type & OUTPUT)
199  return output(n);
200  if (type & PARAMETER && n < parameter.size() && parameter[n].data)
201  return *parameter[n].data;
202  if (type & STATE && n < state.size() && state[n].data)
203  return *state[n].data;
204  if (type & EVENT && n < event.size() && event[n].data)
205  return *event[n].data;
206  if (type & COMMENT && n < comment.size()) {
207  std::istringstream sstr(comment[n].comment);
208  double value;
209  sstr >> value;
210  return value;
211  }
212  return 0.0;
