170 void
171 IO::Block::connect(IO::Block* src,
172  size_t src_num,
173  IO::Block* dest,
174  size_t dest_num)
175 {
176  Mutex::Locker lock(&mutex);
177 
178  if (!src) {
179  ERROR_MSG("Block::connect : invalid source\n");
180  return;
181  }
182  if (src_num >= src->outputs.size()) {
183  ERROR_MSG("Block::connect : invalid source channel\n");
184  return;
185  }
186 
187  if (!dest) {
188  ERROR_MSG("Block::connect : invalid destination\n");
189  return;
190  }
191  if (dest_num >= dest->inputs.size()) {
192  ERROR_MSG("Block::connect : invalid destination channel\n");
193  return;
194  }
195 
196  // Check if the connection exists
197  for (std::list<struct link_t>::const_iterator i =
198  src->outputs[src_num].links.begin();
199  i != src->outputs[src_num].links.end();
200  ++i)
201  if (i->block == dest && i->channel == dest_num) {
202  ERROR_MSG("Block::connect : connection exists\n");
203  return;
204  }
205 
206  struct link_t link;
207 
208  // Make the forward connection (src => dest)
209  link.block = dest;
210  link.channel = dest_num;
211  src->outputs[src_num].links.push_back(link);
212 
213  // Make the backward connection (src <= dest)
214  link.block = src;
215  link.channel = src_num;
216  dest->inputs[dest_num].links.push_back(link);
217 
218  Event::Object event(Event::IO_LINK_INSERT_EVENT);
219  event.setParam("src", src);
220  event.setParam("src_num", &src_num);
221  event.setParam("dest", dest);
222  event.setParam("dest_num", &dest_num);
223  Event::Manager::getInstance()->postEvent(&event);
224 }

238  if (src_num >= src->outputs.size()) {

243  if (!dest) {
244  ERROR_MSG("Block::disconnect : invalid destination\n");
245  return;
246  }
247  if (dest_num >= dest->inputs.size()) {
248  ERROR_MSG("Block::disconnect : invalid destination channel\n");
249  return;
250  }
251 
252  Event::Object event(Event::IO_LINK_REMOVE_EVENT);
253  event.setParam("src", src);
254  event.setParam("src_num", &src_num);
255  event.setParam("dest", dest);
256  event.setParam("dest_num", &dest_num);
257  Event::Manager::getInstance()->postEvent(&event);
258 
259  // Remove the forward connection (src => dest)
260 start_forward_remove:
261  for (std::list<struct link_t>::iterator i =
262  src->outputs[src_num].links.begin();
263  i != src->outputs[src_num].links.end();
264  ++i)
265  if (i->block == dest && i->channel == dest_num) {
266  src->outputs[src_num].links.erase(i);
267  goto start_forward_remove;
268  }
269 
270  // Remove the backward connection (src <= dest)
271 start_backward_remove:
272  for (std::list<struct link_t>::iterator i =
273  dest->inputs[dest_num].links.begin();
274  i != dest->inputs[dest_num].links.end();
275  ++i)
276  if (i->block == src && i->channel == src_num) {
277  dest->inputs[dest_num].links.erase(i);
278  goto start_backward_remove;
279  }
