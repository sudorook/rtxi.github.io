46 Fifo::read(void* buffer, size_t n, bool blocking)
47 {
48  // Acquire the data lock
49  if (blocking)
50  pthread_mutex_lock(&mutex);
51  else if (pthread_mutex_trylock(&mutex) != 0)
52  return 0;
53 
54  // Check that enough data is available
55  if (AVAILABLE < n)
56  if (blocking) {
57  do {
58  pthread_cond_wait(&data_available, &mutex);
59  } while (AVAILABLE < n);
60  } else {
61  pthread_mutex_unlock(&mutex);
62  return 0;
63  }
64 
65  // Copy the data from the fifo
66  if (size - rptr < n) {
67  size_t m = size - rptr;
68  memcpy(buffer, data + rptr, m);
69  memcpy(reinterpret_cast<char*>(buffer) + m, data, n - m);
70  } else
71  memcpy(buffer, data + rptr, n);
72  rptr = (rptr + n) % size;
73 
74  pthread_mutex_unlock(&mutex);
